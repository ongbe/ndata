82 struct kern_cubic {
83  float kern(float x) {
84 
85  assert(fabs(x)<=ONESIDEDWIDTH);
86 
87  float dx = fabs(x);
88  float dx2 = dx*dx;
89  float dx3 = dx2*dx;
90 
91  const float a = -0.5;
92 
93  float convCoeff = 0;
94 
95  if (dx <= 1) {
96  convCoeff = (a+2)*dx3-(a+3)*dx2+1;

102 
103  return convCoeff;
104  };

111 struct interpolate_inner {
112 
113  static
114  T do_it(
115  ndatacontainer<ContainerT, T, ndims> u,
116  vecarray<float, ndims> index_frac
117  )
118  {
119  static_assert(ndims > 1, "Dynamic case not implemented");
120 
121  //u_new loses the.back dimension compared to u
122  //the.back dimension is the one on which the interpolation is performed
123  nvector<T, ndims-1> u_new (u.get_shape().drop_back(), 0);
124 
125  auto ndi_unew = u_new.ndindex(0);
126 
127  //iterating on all the values of the new array u_new
128  //also means iterating on all dimensions (but the last) of the old array
129  //that is we iterate on all the 1D columns of the old array
130  //
131  //iunew_flat is the flattened multidimensional index (with VectorT stride)
132  for (size_t iunew_flat = 0; iunew_flat < u_new.size(); ++iunew_flat) {
133 
134  //setting the matching flatened multidimensional index from the
135  //start of the current column in the old u array
136  //size_t iu0 = 0;
137  auto ndi_u = ndi_unew.append(0);
138 
139  //get a slice from u matching the missing dimension of u_new
140  nvector<T, 1> u_col (make_indexer(u.get_shape().back()), 0);
141  for (size_t iCol = 0; iCol < u_col.size(); ++iCol) {
142  u_col(iCol) = u(ndi_u); //no stride in uOld either since we grab the.back dimension
143  ndi_u.back()++;
144  }
145 
146  //update u_new
147  //recursive call, this one doesn't have subrecursion since ndims == 1
148  u_new(ndi_unew) = interpolate_inner<KernT, 1, ContainerT, T>::do_it(
149  u_col,
150  make_vecarray(index_frac.back())
151  );
152 
153  u_new.increment_ndindex(ndi_unew);
154  }
155 
156  //recurse with one less dimension
157  //will terminate when ndims = 1 (see partially specialized template under)
158  return interpolate_inner<KernT, ndims-1, std::vector<T>, T>::do_it(u_new, index_frac.drop_back());
159  }

198 template<class KernT, long ndims, typename ContainerT, typename T>
199 T interpolate (
200  //flattened array of VectorT
201  //TODO Note on VectorT packing/padding/casting
202  ndatacontainer<ContainerT, T, ndims> u,
203  //sampling is "normalized" by delta so it is expressed in terms of a fraction of
204  //the sourceField indices instead of a real position
205  //[ndims]
206  vecarray<float, ndims> index_frac,
207  //[from 1 to ndims]
208  vecarray<overflow_behaviour, ndims> overflow_behaviours
209  )
210 {
211  auto shape = u.get_shape();
212 
213  assert(
214  index_frac.size() == shape.size()
215  and overflow_behaviours.size() == shape.size()
216  );
217 
218  vecarray<long, ndims> i_starts, i_stops;
219  i_starts = i_stops = vecarray<long, ndims>(shape.dynsize());
220 
221  for (size_t i = 0; i < shape.size(); ++i) {
222 
223  overflow_behaviour ob = overflow_behaviours[i];
224 
225  if (ob == overflow_behaviour::STRETCH) {
226  index_frac[i] = clamp(index_frac[i], float(0), float(shape[i])-1);
227  }
228 
229  long kernWidth = KernT::ONESIDEDWIDTH*2;
230 
231  i_starts[i] = floor(index_frac[i])-long(KernT::ONESIDEDWIDTH)+1;
232  i_stops[i] = i_starts[i] + kernWidth;
233 
234  switch (ob) {
235  case ZERO:
236  i_starts[i] = std::max(0l, i_starts[i]);
237  i_stops[i] = std::min(i_stops[i], long(shape[i]));
238 
239  //early return in case the intersect btw the kernel and the field is empty
240  if(i_starts[i] >= i_stops[i]) return helpers::numtype_adapter<T>::ZERO;
241 
242  assert(i_starts[i] >= 0 and i_stops[i] <= long(shape[i])); //this is not true for other overflow behaviours
243  break;
244  default:
245  //nothing
246  break;
247  }
248 
249  //at least one value in each dimension
250  assert(i_starts[i] < i_stops[i]);
251  }
252 
253  //now extracting the hypercube from u where the kernel is non-zero
254 
255  vecarray<long, ndims> unew_shape = shape;
256 
257  for (size_t idim = 0; idim < ndims; ++idim) {
258  unew_shape[idim] = i_stops[idim] - i_starts[idim];
259  }
260 
261  nvector<T, ndims> unew (unew_shape, helpers::numtype_adapter<T>::ZERO);
262 
263  vecarray<size_t, ndims> ndi_unew = unew.ndindex(0);
264 
265  //iterating on all the values of the new array u_new
266  //and their matching values from the old array
267  //
268  //i is the flattened index in unew
269  for (size_t i = 0; i < unew.size(); ++i) {
270 
271  //compute the multidimensional index for the current value in the old u array
272  //ndi_uold is the multidimensional index in u, which corresponds to i in unew
273  //the overflowBehaviour logic comes here in play
274  vecarray<size_t, ndims> ndi_uold (unew_shape.dynsize());
275 
276  for (size_t idim = 0; idim < ndims; ++idim) {
277  long iUThisDim=(i_starts[idim]+long(ndi_unew[idim]));
278 
279  switch (overflow_behaviours[idim]) {
280  case CYCLIC:
281  iUThisDim = iUThisDim%shape[idim];
282  if(iUThisDim < 0) {
283  iUThisDim += shape[idim];
284  }
285  break;
286  case STRETCH:
287  iUThisDim = clamp(long(iUThisDim), 0l, long(shape[idim])-1l);
288  break;
289  case ZERO:
290  //nothing to do
291  break;
292  default:
293  abort();//shouldn't happen
294  break;
295  }
296 
297  assert(iUThisDim < long(shape[idim]));
298  assert(iUThisDim >= 0);
299  ndi_uold[idim] = size_t(iUThisDim);
300  }
301 
302  unew(ndi_unew) = u(ndi_uold); //no stride in uOld either since we grab the last dimension
303 
304  //update the multidimensional index to match the flat one for the next iteration
305  //needed to locate matching values in the old u array
306  unew.increment_ndindex(ndi_unew);
307  }
308 
309  //adjust indexfrac for new reduced array size
310  for (size_t i = 0; i < index_frac.size(); ++i) {
311  index_frac[i] = index_frac[i] - i_starts[i];
312  }
313 
314  assert(index_frac.size() == unew_shape.size() );
315 
316  return interpolate_inner<KernT, ndims, std::vector<T>, T>::do_it(
317  unew,
318  index_frac
319  );
320 }
